<script>
        // Get the token from the URL
        const urlParams = new URLSearchParams(window.location.search);
        const token = urlParams.get('token');

        if (!token) {
            document.getElementById('message').innerText = 'Invalid or missing token.';
            document.getElementById('resetForm').style.display = 'none';
        }

        document.getElementById('resetForm').addEventListener('submit', async (e) => {
            e.preventDefault();

            const newPassword = document.getElementById('newPassword').value;
            const confirmPassword = document.getElementById('confirmPassword').value;
            const messageDiv = document.getElementById('message');

            if (newPassword !== confirmPassword) {
                messageDiv.innerText = 'Passwords do not match.';
                return;
            }

            try {
                const response = await fetch(address+ '/resetpassword', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/x-www-form-urlencoded',
                    },
                    body: `token=${encodeURIComponent(token)}&newPassword=${encodeURIComponent(newPassword)}`
                });

                const data = await response.json();
                const result = data.Root.DataPage.Record.Result;
                const message = data.Root.DataPage.Record.Message;

                if (result === "true") {
                    messageDiv.style.color = 'green';
                    messageDiv.innerText = message || 'Password reset successfully! Redirecting to login...';
                    setTimeout(() => {
                        window.location.href = address + '/login';
                    }, 2000);
                } else {
                    messageDiv.style.color = 'red';
                    messageDiv.innerText = message || 'Failed to reset password.';
                }
            } catch (error) {
                messageDiv.innerText = 'An error occurred. Please try again.';
            }
        });
    </script>
