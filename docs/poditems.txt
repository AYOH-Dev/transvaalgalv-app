        function renderCategory() {

        function isCategoryComplete() {
            return defectCategories[currentStep].items.every(item => {
                const value = formData[item.key] || item.default;
                if (value !== item.default) {
                    const mitigations = item.mitigations[value] || [];
                    return mitigations.every(mitigation => {
                        const qtyKey = `${item.key}MitigationQty_${mitigation}`;
                        const qty = formData[qtyKey] || '';
                        return qty === '' || (!isNaN(qty) && parseInt(qty) >= 0);
                    });
                }
                return true;
            });
        }

                    defaultMitigations.forEach(mitigation => {
                        formData[`${item.key}MitigationQty_${mitigation}`] = 0;
                    });
                    if (item.default === 'no') {

                        altMitigations.forEach(mitigation => {
                            formData[`${item.key}MitigationQty_${mitigation}`] = 0;
                        });
                    } else {
                        const altValue = Object.keys(item.mitigations).find(key => key !== item.default) || item.default;

                        altMitigations.forEach(mitigation => {
                            formData[`${item.key}MitigationQty_${mitigation}`] = 0;
                        });

                });

            renderCategory();
        }

        function updatePhotoVisibility() {
            defectCategories[currentStep].items.forEach(item => {
                const $defectItem = $(`#defectItems [data-defect="${item.key}"]`);
                const value = formData[item.key];
                const isDefault = value === item.default;
                const $fileInput = $defectItem.find('.file-input');
                const $photoPreview = $defectItem.find('.photo-preview');

                if (!isDefault) {
                    $fileInput.show();
                } else {
                    $fileInput.hide();
                }

                $fileInput.off('change').on('change', function () {
                    const file = this.files[0];
                    console.log(`File selected for ${item.key}:`, file ? file.name : 'none');
                    if (file) {
                        photos[item.key] = file;
                        console.log('Photos after file selection:', photos, Object.keys(photos).length);
                        const reader = new FileReader();
                        reader.onload = function (e) {
                            $photoPreview.attr('src', e.target.result).show();
                        };
                        reader.readAsDataURL(file);
                    } else {
                        delete photos[item.key];
                        $photoPreview.hide();
                    }
                });
            });
        }

        $('#defectItems').on('change', 'input[type="checkbox"]', function () {
            const name = $(this).attr('name');

            if (!formData[name]) formData[name] = [];
            if ($(this).is(':checked')) {
                formData[name] = [...new Set([...(formData[name] || []), value])];
                qtyInput.prop('disabled', false).val(formData[`${name.replace('Mitigation', '')}MitigationQty_${value}`] || 0);
            } else {
                formData[name] = formData[name].filter(v => v !== value);
                qtyInput.prop('disabled', true).val(0);

        function collectFormData() {
            const $row = $('#receiveItemModal').data('clickedRow') || $('.open-details').closest('tr');
            const isNewRow = $row.hasClass('new-row');
            const table = $('#DNItemsTable').DataTable();
            const rowData = table.row($row).data() || [];
            const deliveryNoteNumber = isNewRow ? $row.find('td:eq(1) input').val()?.trim() || '' : rowData[1] || '';
            const lineNumber = rowData[0] || (newRowIndex !== null ? `New_${newRowIndex}` : '');
            const uniqueNumber = rowData[10] || $row.find('.open-details').data('unique-number') || '';
            const dwid = rowData[11] || '';

            const itemSizeElement = document.getElementById('itemSize');
            const itemSizeValue = itemSizeElement ? itemSizeElement.value : $('#receiveItemModal').data('itemSize') || '';

            const expectedQty = parseInt($('#itemQty').val()) || 0;
            const actualQtyReceived = $('#itemQtyReceived').val()?.trim() !== ''? parseInt($('#itemQtyReceived').val()): expectedQty;  
			    
			    
            const quantityDiscrepancy = $('#quantityDiscrepancy').val() || 'none';
            const remainingQty = (quantityDiscrepancy === 'short' && actualQtyReceived < expectedQty)
                ? expectedQty - actualQtyReceived
                : 0;

            const receiveData = {
                documentId: crypto.randomUUID(),
                lineNumber: lineNumber,
                uniqueNumber: uniqueNumber,
                dwid: dwid,
                weighbridgeTicketNumber: $('#weighbridgeTicketNumber').val() || wbt || $('.open-details').data('wbt') || '',
                deliveryNoteNumber: $('#deliveryNoteNumber').val() || deliveryNoteNumber,
                orderNumber: $('#orderNumber').val() || '',
                jobNumber: $('#jobNumber').val() || '',
                other: rowData[6] || '',
                companyName: $('#companyName').val() || '',
                vehicleRegistration: $('#vehicleRegistration').val() || '',
                itemCodeOnDeliveryNote: $('#itemCodeOnDeliveryNote').val() || '',
                itemNameOnDeliveryNote: $('#itemNameOnDeliveryNote').val() || '',
                internalItemName: $('#internalItemName').val() || '',
                itemSize: itemSizeValue,
                itemType: $('#itemType').val() || null,
                process: $('#process').val() || null,
                defectDetected: $('input[name="defectDetected"]:checked').val() || 'No',
                packagingMethod: $('#packagingMethod').val() || null,
                quantityDiscrepancy: quantityDiscrepancy,
                expectedQuantity: expectedQty,
                actualQuantityReceived: actualQtyReceived,
                remainingQuantity: remainingQty,
                materialMarkings: $('#materialMarkings').val() || '',
                materialLength: $('#materialLength').val() || '',
                itemWeight: $('#itemWeight').val() || '',
                fabricator: $('#fabricator').val() || '',
                bay: $('#bay').val() || '',
                accessories: $('#accessories').val() || '',
                comments: $('#comments').val() || ''
            };

            let defectData = {};
            if (receiveData.defectDetected === 'Yes') {
                formData.additionalComments = $('#additionalComments').val();
                defectData = {};
                Object.keys(formData).forEach(key => {
				    const xmlFriendlyKey = key.replace(/\s+/g, '_');
				    const value = formData[key];
				    if (Array.isArray(value) && key.endsWith('Mitigation')) {
				        // Get the base key (e.g. 'holesInadequateMitigation' -> 'holesInadequate')
				        const baseKey = key.replace('Mitigation', '');
				        const item = defectCategories.flatMap(c => c.items).find(i => i.key === baseKey);
				        if (item) {
				            // For every possible mitigation option, output Yes/No
				            Object.values(item.mitigations).flat().filter((v, i, a) => a.indexOf(v) === i).forEach(mitigation => {
				                const mitigationKey = `${xmlFriendlyKey}_${mitigation
				                    .toLowerCase()
				                    .replace(/[^a-z0-9]+/g, '_')
				                    .replace(/^_|_$/g, '')}`;
				                defectData[mitigationKey] = value.includes(mitigation) ? 'Yes' : 'No';
				            });
				        }
				    } else {
				        defectData[xmlFriendlyKey] = value;
				    }
				});
				
				// Add consolidated mitigation fields for DocuWare
				const mitigationFieldMap = {
				    'paint': 'paintMitigation',
				    'damaged': 'damagedMitigation',
				    'rust': 'rustMitigation',
				    'delamination': 'delaminationMitigation',
				    'nonConformingPreGalv': 'nonConformingPreGalvMitigation',
				    'enclosedCavity': 'enclosedCavityMitigation',
				    'threadedArticle': 'threadedArticleMitigation'
				};
				
				Object.entries(mitigationFieldMap).forEach(([defectKey, fieldName]) => {
				    const mitigations = formData[`${defectKey}Mitigation`] || [];
				    if (mitigations.length > 0) {
				        defectData[fieldName] = mitigations.join(', ');
				    } else {
				        defectData[fieldName] = '';
				    }
				});
				
                defectCategories.forEach(category => {
                    category.items.forEach(item => {
                        const mitigations = formData[`${item.key}Mitigation`] || [];
                        mitigations.forEach(mitigation => {
                            const qtyKey = `${item.key}MitigationQty_${mitigation}`;
                            const xmlFriendlyQtyKey = qtyKey.replace(/\s+/g, '_');
                            if (formData[qtyKey]) {
                                defectData[xmlFriendlyQtyKey] = formData[qtyKey];
                            }
                        });
                    });
                });
            }

            return { receiveInfo: receiveData, defectInfo: defectData, timestamp: new Date().toISOString() };
        }

                    isNewRow: newRowIndex !== null // Add flag for backend
                }
            };
            console.log('JSON Payload:', JSON.stringify(jsonPayload, null, 2));
            fetch(address + '/submitItemData', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'Accept': 'application/json'
                },
                body: JSON.stringify(jsonPayload)
            })
			.then(async response => {
			    console.log('Submit response status:', response.status, response.statusText);
			    if (!response.ok) {
			        const errorText = await response.text();
			        console.error('Submit response body:', errorText);
			        throw new Error(`Server Error (${response.status}): ${errorText || response.statusText}`);
			    }
			    
			    // Handle "No documents found" response
			    const responseText = await response.text();
			    if (responseText.includes('No documents found')) {
			        console.log('Server returned "No documents found" - treating as success');
			        // Return a mock successful response
			        return {
			            Field: [{
			                FieldName: 'DWDOCID',
			                Item: 'TEMP_' + Date.now() // Temporary document ID
			            }]
			        };
			    }
			    
			    // Parse as JSON for normal responses
			    try {
			        return JSON.parse(responseText);
			    } catch (e) {
			        console.error('Failed to parse response as JSON:', responseText);
			        throw new Error('Invalid JSON response from server');
			    }
			})
                .then(data => {
                    console.log('Full server response:', JSON.stringify(data, null, 2));
                    const docField = data.Field?.find(field => field.FieldName === 'DWDOCID');
                    if (!docField || !docField.Item) {
                        console.error('DWDOCID missing or invalid:', { dataField: data.Field });
                        throw new Error('Invalid response format or missing DWDOCID');
                    }
                    const documentId = docField.Item;
                    // Update dwid for new row in DataTable
                    if (newRowIndex !== null) {
                        const table = $('#DNItemsTable').DataTable();
                        let rowNode = null;
                        table.rows().every(function () {
                            if (this.data()[0] === newRowIndex) {
                                rowNode = this.node();
                                const rowData = this.data();
                                rowData[11] = documentId; // Update dwid
                                rowData[8] = '<span class="text-success">Received</span>'; // Update Receive column
                                this.data(rowData);
                            }
                        });
                        if (rowNode) {
                            $(rowNode).find('.open-details').prop('disabled', true); // Disable Receive button
                            $(rowNode).removeClass('new-row table-danger'); // Remove new-row styling
                        }
                    }
                    if (Object.keys(photos).length > 0) {
                        return uploadImages(documentId, formData.receiveInfo.uniqueNumber, photos)
                            .then(uploadResult => {
                                photos = {}; // Clear photos only after successful upload
                                return { success: true, documentId };
                            });
                    }
                    photos = {}; // Clear photos if no upload is needed
                    return { success: true, documentId };
                })
				.then(() => {
				    console.log('Submission completed successfully at:', new Date().toLocaleString('en-ZA', { timeZone: 'Africa/Johannesburg' }));
				    
				    // For new rows, remove them from the table (simulate normal receive behavior)
				    if (newRowIndex !== null) {
				        const table = $('#DNItemsTable').DataTable();
				        let rowToRemove = null;
				        table.rows().every(function () {
				            if (this.data()[0] === newRowIndex.toString()) {
				                rowToRemove = this;
				                return false; // Stop iteration
				            }
				        });
				        if (rowToRemove) {
				            rowToRemove.remove();
				            table.draw();
				            console.log('Removed new row after successful submission');
				        }
				    }
    
    $('#receiveItemModal').modal('hide');
    $('#passedModal').modal('hide');
    $('body').removeClass('modal-open');
    $('.modal-backdrop').remove();
    resetForms();
    
    // Don't refresh table to avoid "No documents found" error
    // refreshTable();
    
    $('#statusBar').text('Submission successful - Item received');
})
                .catch(error => {
                    console.error('Detailed error information:', {
                        message: error.message,
                        stack: error.stack,
                        timestamp: new Date().toLocaleString('en-ZA', { timeZone: 'Africa/Johannesburg' })
                    });
                    $('#statusBar').text(`Error: ${error.message}`);
                    alert(`Submission failed: ${error.message}\nPlease check the console for more details.`);
                })
                .finally(() => {
                    $('#loader').fadeOut();
                    if ($('.modal-backdrop').length > 0) {
                        $('.modal-backdrop').remove();
                        $('body').removeClass('modal-open');
                    }
                });
        }

        function uploadImages(documentId, uniqueNumber, photos) {
            console.log('Uploading images at:', new Date().toLocaleString('en-ZA', { timeZone: 'Africa/Johannesburg' }), { documentId, uniqueNumber, photoKeys: Object.keys(photos) });
            const photoFormData = new FormData();
            Object.entries(photos).forEach(([defectKey, file]) => {
                const xmlFriendlyKey = defectKey.replace(/\s+/g, '_');
                const fileName = `${uniqueNumber}_${xmlFriendlyKey}_${documentId}.jpg`;
                photoFormData.append(`photo_${xmlFriendlyKey}`, file, fileName);
            });
            photoFormData.append('documentId', documentId);
            photoFormData.append('uniqueNumber', uniqueNumber);
            photoFormData.append('timestamp', new Date().toISOString());
            photoFormData.append('user', userData?.username || 'unknown');
            console.log('Photo FormData contents:');
            for (let [key, value] of photoFormData.entries()) {
                console.log(`  ${key}: ${value instanceof File ? value.name : value}`);
            }
            return fetch(address + '/uploadImages', {
                method: 'POST',
                body: photoFormData
            })
                .then(async response => {
                    console.log('Upload response status:', response.status, response.statusText);
                    if (!response.ok) {
                        const errorText = await response.text();
                        console.error('Upload response body:', errorText);
                        throw new Error(`Image upload failed (${response.status}): ${errorText || response.statusText}`);
                    }
                    return response.json();
                })
                .then(result => {
                    console.log('Image upload successful:', result, 'Timestamp:', new Date().toLocaleString('en-ZA', { timeZone: 'Africa/Johannesburg' }));
                    $('#statusBar').text('Images Successfully Uploaded');
                    setTimeout(() => $('#statusBar').text('System Ready'), 3000);
                    return result;
                })
                .catch(error => {
                    console.error('Image upload error:', {
                        message: error.message,
                        stack: error.stack,
                        timestamp: new Date().toLocaleString('en-ZA', { timeZone: 'Africa/Johannesburg' })
                    });
                    $('#statusBar').text(`Error Uploading Images: ${error.message}`);
                    alert(`Image upload failed: ${error.message}\nPlease check the console for details.`);
                    throw error;
                });
        }

		function validateReceiveForm() {
		    let expectedQty = parseInt($('#receiveItemModal').data('expectedQty')) || parseInt($('#itemQty').val()) || 0;
		    $('#itemQty').val(expectedQty);
		    let isValid = true;
		    let errorMessages = [];
		    const fields = {
		        itemCode: { id: '#itemCodeOnDeliveryNote', message: 'Item Code on Delivery Note is required.' },
		        itemName: { id: '#itemNameOnDeliveryNote', message: 'Item Name on Delivery Note is required.' },
		        internalItemName: { id: '#internalItemName', message: 'Internal Item Name is required.' },
		        itemType: { id: '#itemType', message: 'Item Type is required.' },
		        process: { id: '#process', message: 'Process is required.' },
		        packagingMethod: { id: '#packagingMethod', message: 'Packaging Method is required.' }
		    };
		    const quantityDiscrepancy = $('#quantityDiscrepancy').val() || 'none';
		    const actualQtyReceived = parseInt($('#itemQtyReceived').val());
		    const defectDetected = $('input[name="defectDetected"]:checked').val() || '';
		    const itemSize = $('#itemSize').val()?.trim() || '';
		    const bay = $('#bay').val()?.trim() || '';
		
		    Object.values(fields).forEach(field => {
		        const value = $(field.id).val()?.trim() || '';
		        if (!value) {
		            isValid = false;
		            errorMessages.push(field.message);
		            $(field.id).addClass('is-invalid').next('.invalid-feedback').text(field.message);
		        } else {
		            $(field.id).removeClass('is-invalid').next('.invalid-feedback').text('');
		        }
		    });
		
		    if (!expectedQty || isNaN(expectedQty) || expectedQty <= 0) {
		        isValid = false;
		        errorMessages.push('Expected Quantity must be a positive number.');
		        $('#itemQty').addClass('is-invalid').next('.invalid-feedback').text('Expected Quantity must be positive.');
		    } else {
		        $('#itemQty').removeClass('is-invalid').next('.invalid-feedback').text('');
		    }
		
		    
		    if (quantityDiscrepancy !== 'none') {
		        if (isNaN(actualQtyReceived) || actualQtyReceived < 0) {
		            isValid = false;
		            errorMessages.push('Actual Quantity Received must be 0 or greater when there is a discrepancy.');
		            $('#itemQtyReceived').addClass('is-invalid').next('.invalid-feedback').text('Actual Quantity must be 0 or greater.');
		        } else {
		            const difference = actualQtyReceived - expectedQty;
		            
		            if (quantityDiscrepancy === 'short' && difference >= 0) {
		                isValid = false;
		                errorMessages.push("Actual Quantity must be less than Expected Quantity for 'Short Supplied'.");
		                $('#itemQtyReceived').addClass('is-invalid').next('.invalid-feedback').text('Actual Quantity must be less than Expected for Short Supplied.');
		            } else if (quantityDiscrepancy === 'over' && difference <= 0) {
		                isValid = false;
		                errorMessages.push("Actual Quantity must be greater than Expected Quantity for 'Over Supplied'.");
		                $('#itemQtyReceived').addClass('is-invalid').next('.invalid-feedback').text('Actual Quantity must be greater than Expected for Over Supplied.');
		            } else if (Math.abs(difference) > expectedQty * 0.1) {
		                // Warning only, doesn't block submission
		                errorMessages.push('Warning: Difference exceeds 10% of Expected Quantity.');
		            } else {
		                $('#itemQtyReceived').removeClass('is-invalid').next('.invalid-feedback').text('');
		            }
		        }
		    } else {
		        $('#itemQtyReceived').removeClass('is-invalid').next('.invalid-feedback').text('');
		    }
		
		    if (!defectDetected) {
		        isValid = false;
		        errorMessages.push('Defect Detected selection (Yes/No) is required.');
		        $('input[name="defectDetected"]').addClass('is-invalid').next('.invalid-feedback').text('Defect Detected is required.');
		    } else {
		        $('input[name="defectDetected"]').removeClass('is-invalid').next('.invalid-feedback').text('');
		    }
		
		    if (!itemSize) {
		        isValid = false;
		        errorMessages.push('Item Size is required.');
		        $('#itemSize').addClass('is-invalid').next('.invalid-feedback').text('Item Size is required.');
		    } else {
		        $('#itemSize').removeClass('is-invalid').next('.invalid-feedback').text('');
		    }
		
		    console.log('Bay value during validation:', bay);
		
		    $('#receiveItemBtn').prop('disabled', !isValid);
		    const errorContainer = $('#receiveFormErrors');
		    errorContainer.html(isValid ? '' : errorMessages.map(msg => `<div class="text-danger">${msg}</div>`).join('')).toggleClass('d-none', isValid);
		    return isValid;
		}

		}

        function updateProcessOptions() {
            const itemType = $('#itemType').val();
            const $process = $('#process');
            const originalValue = $process.val();
            $process.find('option').prop('disabled', false);
            if (itemType === 'galvanised') {
                $process.find('option[value="galvanising"]').prop('disabled', true).attr('title', 'Not available for Material Type: Galvanised');
                $process.find('option[value="galvanising_paint"]').prop('disabled', true).attr('title', 'Not available for Material Type: Galvanised');
                $process.find('option[value="shotblast_galvanising"]').prop('disabled', true).attr('title', 'Not available for Material Type: Galvanised');
                $process.val('strip_regalvanise');
            } else if (itemType === 'blacksteel') {
                $process.find('option[value="strip_only"]').prop('disabled', true).attr('title', 'Not available for Material Type: Black Steel');
                $process.find('option[value="strip_regalvanise"]').prop('disabled', true).attr('title', 'Not available for Material Type: Black Steel');
                $process.find('option[value="strip_galvanising_paint"]').prop('disabled', true).attr('title', 'Not available for Material Type: Black Steel');
                $process.find('option[value="shotblast_strip_regalvanising"]').prop('disabled', true).attr('title', 'Not available for Material Type: Black Steel');
                $process.find('option[value="shotblast_strip_regalvanising_paint"]').prop('disabled', true).attr('title', 'Not available for Material Type: Black Steel');
                if ($process.find(`option[value="${originalValue}"]`).prop('disabled')) {
                    $process.val('');
                }
            } else {
                $process.val('');
            }
            validateReceiveForm();
        }

        window.toggleQuantityFields = function () {
            const discrepancy = $('#quantityDiscrepancy').val();
            const $quantityFields = $('#quantityFields');
            const $qtyDifferenceHelp = $('#qtyDifferenceHelp');
            let expectedQty = parseInt($('#receiveItemModal').data('expectedQty')) || 0;
            if (!expectedQty) {
                const row = $('#receiveItemModal').data('clickedRow');
                expectedQty = parseInt(row.find('td:eq(5)').text().trim()) || 0;
                $('#receiveItemModal').data('expectedQty', expectedQty);
            }
            const actualQty = parseInt($('#itemQtyReceived').val()) || expectedQty;
            const $itemQty = $('#itemQty');
            const $itemQtyDom = document.getElementById('itemQty');
            if ($itemQtyDom) {
                $itemQtyDom.value = expectedQty;
                $itemQty.val(expectedQty);
                $itemQtyDom.setAttribute('value', expectedQty);
                $itemQty.trigger('input');
                $itemQtyDom.style.display = 'none';
                setTimeout(() => {
                    $itemQtyDom.style.display = 'block';
                    $itemQtyDom.focus();
                }, 0);
            }
            if (discrepancy === 'none') {
                $quantityFields.addClass('d-none').removeClass('alert-warning');
                $('#itemQtyReceived').val(expectedQty);
            } else {
                $quantityFields.removeClass('d-none');
                $('#itemQtyReceived').val(actualQty);
                if (actualQty !== expectedQty) {
                    $quantityFields.addClass('alert-warning');
                    const difference = actualQty - expectedQty;
                    $qtyDifferenceHelp.text(`Difference: ${difference} units (${discrepancy === 'short' ? 'Short' : 'Over'} Supplied)`);
                } else {
                    $quantityFields.removeClass('alert-warning');
                    $qtyDifferenceHelp.empty();
                }
            }
            validateReceiveForm();
        };
