<script>
        $(document).ready(function () {
            // Enter key triggers login
            $(document).keypress(function (event) {
                if (event.which === 13) {
                    $("#LogInForm").submit();
                }
            });
			
			// Set a maximum loader timeout
		    let loaderTimeout;
		    
		    // Override show to set timeout
		    const originalShow = $.fn.show;
		    const originalFadeIn = $.fn.fadeIn;
		    
		    $.fn.show = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		            loaderTimeout = setTimeout(() => {
		                console.error('LOADER TIMEOUT: Forcing loader to hide after 30 seconds');
		                $('#loader').hide();
		                $('#statusBar').text('Operation timed out - please refresh');
		            }, 30000); // 30 seconds
		        }
		        return originalShow.apply(this, arguments);
		    };
		    
		    $.fn.fadeIn = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		            loaderTimeout = setTimeout(() => {
		                console.error('LOADER TIMEOUT: Forcing loader to hide after 30 seconds');
		                $('#loader').fadeOut();
		                $('#statusBar').text('Operation timed out - please refresh');
		            }, 30000);
		        }
		        return originalFadeIn.apply(this, arguments);
		    };
		    
		    // Clear timeout when hidden
		    const originalHide = $.fn.hide;
		    const originalFadeOut = $.fn.fadeOut;
		    
		    $.fn.hide = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		        }
		        return originalHide.apply(this, arguments);
		    };
		    
		    $.fn.fadeOut = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		        }
		        return originalFadeOut.apply(this, arguments);
		    };
            // Base address for API calls
            //const address = 'http://presentationvm:8089';

            // Login form submission
            $("#LogInForm").submit(function (e) {
                e.preventDefault();
                const username = $("#LoginName").val().toLowerCase().trim();
                const password = $("#LoginPassword").val();
                const rememberMe = $("#rememberMe").is(":checked");

                if (!username) {
                    $("#feedback").text("Please enter your username.");
                    return;
                }
                if (!password) {
                    $("#feedback").text("Please enter your password.");
                    return;
                }

                $("#loader").fadeIn();
                $("#feedback").text("");
                $("#LogInForm button[type='submit']").prop("disabled", true); // Disable button

                $.ajax({
                    url: address + '/authenticate',
                    type: 'POST',
                    data: { LoginName: username, LoginPassword: password },
                    success: function (data) {
                        try {
                            const results = JSON.parse(data);
                            const result = results.Root.DataPage.Record.Result;
                            const roleName = results.Root.DataPage.Record.RoleName;
                            const usernameFromResponse = results.Root.DataPage.Record.Username; // Get Username from response

                            if (result === "true" && roleName) {
                                // Create a user object to store in storage
                                const user = {
                                    role: roleName,
                                    username: usernameFromResponse // Use server-provided Username
                                };
                                const storage = rememberMe ? localStorage : sessionStorage;
                                storage.setItem("user_data", JSON.stringify(user));
                                $("#feedback").text("Logging in...");
                                switch (roleName) {
                                    case "Admin":
                                        //window.location.href = `${address}/admin_dashboard`;
                                        window.location.href = address + '/index';
                                        break;
                                    case "Receiver":
                                    case "POD Creator":
                                    default:
                                        //window.location.href = `${address}/main_dashboard`;
                                        window.location.href = address + '/index';
                                }
                            } else {
                                $("#feedback").text("Login failed. Check your credentials.");
                                $("#loader").fadeOut();
                                $("#LogInForm button[type='submit']").prop("disabled", false); // Re-enable button
                            }
                        } catch (err) {
                            $("#feedback").text("Error parsing response. Contact support.");
                            $("#loader").fadeOut();
                            $("#LogInForm button[type='submit']").prop("disabled", false); // Re-enable button
                            console.error("Parse error:", err);
                        }
                    },
                    error: function (jqXHR, textStatus, errorThrown) {
                        $("#feedback").text("Server error. Please try again later.");
                        $("#loader").fadeOut();
                        $("#LogInForm button[type='submit']").prop("disabled", false); // Re-enable button
                        console.error("AJAX error:", textStatus, errorThrown);
                    }
                });
            });

            // Forgot password form submission
            $("#recoveryForm").submit(function (e) {
                e.preventDefault();
                const email = $("#recoveryEmail").val().trim();

                if (!email) {
                    $("#recoveryFeedback").text("Please enter your email.");
                    return;
                }

                $("#loader").fadeIn();
                $("#recoveryFeedback").text("Sending reset link...");

                $.ajax({
                    url: `${address}/requestPasswordReset`,
                    type: 'POST',
                    data: { email: email },
                    success: function (data) {
                        try {
                            const results = JSON.parse(data);
                            const success = results.Root.DataPage.Record.Result === "true";

                            if (success) {
                                $("#recoveryFeedback").text("Check your email for a reset link.");
                            } else {
                                $("#recoveryFeedback").text("Email not found.");
                            }
                            $("#loader").fadeOut();
                        } catch (err) {
                            $("#recoveryFeedback").text("Error parsing response. Contact support.");
                            $("#loader").fadeOut();
                            console.error("Parse error:", err);
                        }
                    },
                    error: function () {
                        $("#recoveryFeedback").text("Error sending reset link. Try again.");
                        $("#loader").fadeOut();
                    }
                });
            });

            // Initial setup
            $("#loader").fadeOut();
        });
