<script>
        $(document).ready(function () {
            // Check session
            const userData = checkSession();
            if (!userData) return;

            const role = userData.role || '';
            
            // Set a maximum loader timeout
		    let loaderTimeout;
		    
		    // Override show to set timeout
		    const originalShow = $.fn.show;
		    const originalFadeIn = $.fn.fadeIn;
		    
		    $.fn.show = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		            loaderTimeout = setTimeout(() => {
		                console.error('LOADER TIMEOUT: Forcing loader to hide after 30 seconds');
		                $('#loader').hide();
		                $('#statusBar').text('Operation timed out - please refresh');
		            }, 30000); // 30 seconds
		        }
		        return originalShow.apply(this, arguments);
		    };
		    
		    $.fn.fadeIn = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		            loaderTimeout = setTimeout(() => {
		                console.error('LOADER TIMEOUT: Forcing loader to hide after 30 seconds');
		                $('#loader').fadeOut();
		                $('#statusBar').text('Operation timed out - please refresh');
		            }, 30000);
		        }
		        return originalFadeIn.apply(this, arguments);
		    };
		    
		    // Clear timeout when hidden
		    const originalHide = $.fn.hide;
		    const originalFadeOut = $.fn.fadeOut;
		    
		    $.fn.hide = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		        }
		        return originalHide.apply(this, arguments);
		    };
		    
		    $.fn.fadeOut = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		        }
		        return originalFadeOut.apply(this, arguments);
		    };
		    
		    
		    $(document).ajaxError(function(event, jqXHR, ajaxSettings, thrownError) {
		        if (!jqXHR.isHandled) {
		            console.error('Global AJAX error caught:', {
		                url: ajaxSettings.url,
		                status: jqXHR.status,
		                statusText: jqXHR.statusText,
		                error: thrownError,
		                timestamp: new Date().toISOString()
		            });
		            
		            $('#loader').hide();
		            $('#statusBar').text('Network error occurred');
		            $("#errorMessage").removeClass('d-none').text('Network error - please try again.');
		        }
		    });
            
            $("#loader").show();
            $("#pagename").text("Pending PODs");
            $("#statusBar").text('Fetching Data...');

            // Role-based access
            if (role !== 'Admin') {
                $("#navSettings").hide();
            }
            
			

            getData();

            $('#refreshButton').click(function () {
                $("#refreshButton").prop('disabled', true).text('Refreshing...');
                $("#loader").show();
                var table = $('#deliveryNotesTable').DataTable();
                table.destroy();
                $("#row_openPODs").empty();
                $("#errorMessage").addClass('d-none');
                $("#noDataMessage").addClass('d-none');
                getData();
            });

            $('#retryButton').click(function () {
                $("#errorMessage").addClass('d-none');
                $("#loader").show();
                getData();
            });

            $('#createPodButton').click(function () {
                $("#loader").fadeIn();
                $("#statusBar").text('Navigating to Create New POD...');
                window.location.href = address + '/newPod';
            });

            function getData() {
                $.ajax({
                    url: address + '/getTData',
                    type: 'POST',
                    dataType: 'json',
                    success: function (data) {
                        try {
                            console.log('Raw response:', data); // Debug raw response

                            let results = data;
                            if (typeof data === 'string') {
                                try {
                                    results = JSON.parse(data);
                                } catch (e) {
                                    console.error('Error parsing JSON:', e);
                                    throw new Error('Invalid JSON response');
                                }
                            }

                            if (results && results.Root && results.Root.DataPage) {
                                let dataPage = results.Root.DataPage;
                                let records = [];

                                if (Array.isArray(dataPage)) {
                                    records = dataPage.map(dp => dp.Record);
                                } else if (dataPage.Record) {
                                    records = Array.isArray(dataPage.Record) ? dataPage.Record : [dataPage.Record];
                                }

                                if (records.length > 0) {
                                    console.log('Processing', records.length, 'records');
                                    let items = '';
                                    records.forEach(record => {
                                        const weighbridgeTicket = record.WeighbridgeTicketNo || '';
                                        const deliveryNote = record.DeliveryNoteNo || '';
                                        const company = record.Company || '';
                                        const orderNo = record.OrderNo || '';
                                        const processor = record.Processor || '';

                                        items += `<tr>
                                            <td>${weighbridgeTicket}</td>
                                            <td>${deliveryNote}</td>
                                            <td>${company}</td>
                                            <td>${orderNo}</td>
                                            <td>${processor}</td>
                                            <td>
                                                <button class='btn btn-success open-details' 
                                                    data-wbt='${weighbridgeTicket}'
                                                    data-company='${company}' 
                                                    type='button'
                                                    aria-label='Open details for ${deliveryNote}'>Open</button>
                                            </td>
                                        </tr>`;
                                    });

                                    $("#row_openPODs").append(items);

                                    $('#deliveryNotesTable').DataTable({
                                        columnDefs: [{
                                            targets: -1,
                                            orderable: false
                                        }],
                                        language: {
                                            aria: {
                                                sortAscending: ': activate to sort column ascending',
                                                sortDescending: ': activate to sort column descending'
                                            }
                                        }
                                    });

                                    $('#deliveryNotesTable').on('click', '.open-details', function () {
                                        const wbt = $(this).data('wbt');
                                        const company = $(this).data('company');
                                        $("#loader").fadeIn();
                                        $("#statusBar").text(`Navigating to Delivery Note Details...`);
                                        window.location.href = address + `/podsitems?wbt=${encodeURIComponent(wbt)}&company=${encodeURIComponent(company)}`;
                                    });

                                    $("#loader").fadeOut();
                                    $("#statusBar").text('System Ready');
                                } else {
                                    $("#noDataMessage").removeClass('d-none');
                                    $("#loader").fadeOut();
                                    $("#statusBar").text('System Ready');
                                }
                            } else {
                                $("#noDataMessage").removeClass('d-none');
                                $("#loader").fadeOut();
                                $("#statusBar").text('System Ready');
                            }
                        } catch (e) {
                            console.error('Error processing response:', e);
                            $("#errorMessage").removeClass('d-none').text(`Error loading data: ${e.message}.`);
                            $("#loader").fadeOut();
                            $("#statusBar").text('Error');
                        }
                        $("#refreshButton").prop('disabled', false).text('Refresh');
                    },
                    error: function (xhr, status, error) {
                    	xhr.isHandled = true;
                        console.error('AJAX error:', status, error, xhr.responseText);
                        $("#errorMessage").removeClass('d-none').text(`Error loading data: ${xhr.statusText || 'Unknown error'}.`);
                        $("#loader").fadeOut();
                        $("#statusBar").text('Error');
                        $("#refreshButton").prop('disabled', false).text('Refresh');
                    }
                });
            }
        });
    </script>
