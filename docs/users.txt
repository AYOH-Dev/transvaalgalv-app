            function initializeApp() {

                const userData = checkSession();

                function fetchRoles() {
                    return $.ajax({
                        url: address+'/getRoles',
                        type: 'GET',
                        success: function (response) {
                            console.log('Fetched roles (raw):', response);
                            let rolesData;
                            try {
                                if (typeof response === 'string') {
                                    rolesData = JSON.parse(response);
                                } else {
                                    rolesData = response;
                                }

                                if (rolesData.Root && rolesData.Root.DataPage) {
                                    roles = rolesData.Root.DataPage.map(item => item.Record);
                                    console.log('Extracted roles:', roles);
                                    populateRoleDropdown(roles);
                                } else {
                                    throw new Error('Invalid roles data structure');
                                }
                            } catch (e) {
                                console.error('Error parsing roles:', e);
                                $("#statusBar").text('Error loading roles.');
                                roles = [];
                            }
                        },
                        error: function (xhr, status, error) {
                        	xhr.isHandled = true;
                            console.error('Error fetching roles:', status, error);
                            $("#statusBar").text('Error loading roles.');
                            roles = [];
                        }
                    });
                }

                    } else {

                        $('#lastName').addClass('is-invalid').after('<div class="invalid-feedback">Last Name is required.</div>');

                        isValid = false;

                        $('#role').addClass('is-invalid').after('<div class="invalid-feedback">Role is required.</div>');
                        isValid = false;

                    if (!isValid) return;

                        UserID: userId,
                        Username: username,
                        FirstName: firstName,
                        LastName: lastName,
                        Email: email,
                        RoleID: roleId,
                        IsActive: isActive,
                        Operation: userId ? "Edit" : "Add"
                    };
