			            ITEM_SIZE: $(this).find('input[name="itemSize"]').val() || '',
			            ITEM_QUANTITY: $(this).find('input[name="itemQuantity"]').val() || '',
			            WEIGHT: $(this).find('input[name="itemWeight"]').val() || '',
			            MATERIAL_MARKINGS: $(this).find('input[name="materialMarkings"]').val() || '',
			            MATERIAL_LENGTH: $(this).find('input[name="materialLength"]').val() || '',
			            JOB_NUMBER: $(this).find('input[name="jobNumber"]').val() || '',
			            OTHER: $(this).find('input[name="other"]').val() || ''
			        });
			    });
			
			    var submit_url = address + '/newPODSubmit';
			    console.log('Submit URL:', submit_url);
			    console.log('Form Data:', JSON.stringify(formData));
			
			    $.ajax({
			        url: submit_url,
			        type: 'POST',
			        data: JSON.stringify(formData),
			        contentType: 'application/json',
			        dataType: 'json',
			        success: function (response) {
			            console.log('POD submission response:', response);
			            let result = response;
			
			            if (typeof response === 'string') {
			                try {
			                    result = JSON.parse(response);
			                } catch (e) {
			                    console.error('Error parsing response as JSON:', e);
			                    $("#loader").hide();
			                    $("#statusBar").text('Error Creating POD: Invalid response format from server.');
			                    $submitBtn.prop('disabled', false); // ADD THIS: Re-enable on error
			                    return;
			                }
			            }
			
			            if (result.DWStatus && result.DWStatus.status === 200) {
			                $("#loader").hide();
			                $("#statusBar").text('POD Created Successfully');
			                formChanged = false;
			                // Keep button disabled during redirect
			                setTimeout(() => {
			                    window.location.href = address + '/pods';
			                }, 1000);
			            } else {
			                $("#loader").hide();
			                $("#statusBar").text('Error Creating POD: ' + (result.message || 'Unknown error'));
			                $submitBtn.prop('disabled', false); // ADD THIS: Re-enable on error
			            }
			        },
			        error: function (xhr, status, error) {
			        	xhr.isHandled = true;
			            $("#loader").hide();
			            $("#statusBar").text('Error Creating POD: ' + (xhr.statusText || 'Unknown error'));
			            console.error('Error:', status, error);
			            $submitBtn.prop('disabled', false); // ADD THIS: Re-enable on error
			        }
			    });
			});

            // Cancel button functionality
            $('#cancelButton').click(function () {
                if (formChanged) {
                    if (confirm('You have unsaved changes. Are you sure you want to cancel?')) {
                        window.location.href = address + '/index';
                    }
                } else {
                    window.location.href = address + '/index';
                }
            });
