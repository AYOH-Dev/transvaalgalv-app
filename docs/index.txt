<script>
        $(document).ready(function () {
            // Check session
            const userData = checkSession();
            if (!userData) return; // Stop execution if no session

            const username = userData.username;
            const role = userData.role || '';

            // Display "Welcome, Username" in the navbar
            const welcomeMessage = $("#welcomeMessage");
            if (username) {
                welcomeMessage.text(`Welcome, ${username}`);
            } else {
                welcomeMessage.text('Welcome, Guest');
            }

            // Handle logout
            $("#navLogout").click(function (e) {
                e.preventDefault();
                // Clear user_data from both localStorage and sessionStorage
                localStorage.removeItem("user_data");
                sessionStorage.removeItem("user_data");
                window.location.href = 'login';
            });
            
            // Set a maximum loader timeout
		    let loaderTimeout;
		    
		    // Override show to set timeout
		    const originalShow = $.fn.show;
		    const originalFadeIn = $.fn.fadeIn;
		    
		    $.fn.show = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		            loaderTimeout = setTimeout(() => {
					    console.error('LOADER TIMEOUT: Forcing loader to hide after 30 seconds');
					    $('#loader').removeClass('d-flex').addClass('d-none');
					    $('#statusBar').text('Operation timed out - please refresh');
					}, 30000); // 30 seconds
		        }
		        return originalShow.apply(this, arguments);
		    };
		    
		    $.fn.fadeIn = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		            loaderTimeout = setTimeout(() => {
					    console.error('LOADER TIMEOUT: Forcing loader to hide after 30 seconds');
					    $('#loader').removeClass('d-flex').addClass('d-none');
					    $('#statusBar').text('Operation timed out - please refresh');
					}, 30000); // 30 seconds
		        }
		        return originalFadeIn.apply(this, arguments);
		    };
		    
		    // Clear timeout when hidden
		    const originalHide = $.fn.hide;
		    const originalFadeOut = $.fn.fadeOut;
		    
		    $.fn.hide = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		        }
		        return originalHide.apply(this, arguments);
		    };
		    
		    $.fn.fadeOut = function() {
		        if (this.attr('id') === 'loader') {
		            clearTimeout(loaderTimeout);
		        }
		        return originalFadeOut.apply(this, arguments);
		    };
		    
	        $(document).ajaxError(function(event, jqXHR, ajaxSettings, thrownError) {
		        if (!jqXHR.isHandled) {
		            console.error('Global AJAX error caught:', {
		                url: ajaxSettings.url,
		                status: jqXHR.status,
		                statusText: jqXHR.statusText,
		                error: thrownError,
		                timestamp: new Date().toISOString()
		            });
		            
		            $('#loader').removeClass('d-flex').addClass('d-none');
		            $('#statusBar').text('Network error occurred');
		        }
		    });												

            // Debugging: Log the loader element
            console.log("Loader element:", $("#loader").length, $("#loader").is(":visible"));

            $("#loader").removeClass('d-none').addClass('d-flex').show();
            $("#statusBar").text('Fetching Data...');

            // Debugging: Confirm loader is shown
            console.log("After show - Loader visible:", $("#loader").is(":visible"));

            const urlParams = new URLSearchParams(window.location.search);
            const wbt = urlParams.get('wbt') || '';
            if (wbt) {
                $('title').text(`Transvaal Galvanisers - Goods Receiving (${wbt})`);
            }

            if (role !== 'Admin') {
                if (role !== 'Receiver') {
                    $("#receive-goods-card").hide();
                }
                if (role !== 'POD Creator') {
                    $("#new-pod-card").hide();
                }
                if (role !== 'Admin') {
                    $("#settings-card").hide();
                }
            }

            $.ajax({
			    url: address + '/getLandingData',
			    type: "POST",
			    
			    // ADD THIS DEBUGGING BLOCK:
			    beforeSend: function() {
			        console.log('🔵 AJAX REQUEST STARTING:', {
			            url: address + '/getLandingData',
			            timestamp: new Date().toISOString()
			        });
			    },
			    
			    complete: function(xhr, status) {
			        console.log('🟢 AJAX REQUEST COMPLETED:', {
			            status: status,
			            httpStatus: xhr.status,
			            timestamp: new Date().toISOString()
			        });
			    },
			    
			    timeout: 15000, // ADD 15 second timeout
			    
			    success: function (response) {
			        console.log('✅ SUCCESS HANDLER CALLED');
			        console.log('Response:', response);
			        
			        try {
			            var results = JSON.parse(response);
			            console.log("Parsed Results:", results);
			
			            $("#pending-receipts .stat-number").text(results.Pending || '0');
			            $("#completed-today .stat-number").text(results.Completed || '0');
			            $("#accuracy-rate .stat-value").text(results.AccuracyRate ? results.AccuracyRate + '%' : 'N/A');
			            $("#quality-checks .stat-value").text(results.QualityChecks || 'N/A');
			
			            console.log('🎯 About to hide loader...');
						$("#loader").removeClass('d-flex').addClass('d-none');
						console.log('✅ Loader hidden');
			            
			            $("#statusBar").text('System Ready');
			            $("#startReceiving, #openSettings, #openPOD").prop('disabled', false);
			        } catch (err) {
			            console.error("❌ Error in success callback:", err);
			            $("#loader").removeClass('d-flex').addClass('d-none');
			            $("#statusBar").text('Error parsing data.');
			            $("#startReceiving, #openSettings, #openPOD").html('Retry').prop('disabled', false).off('click').on('click', function () {
			                location.reload();
			            });
			        }
			    },
			    error: function (xhr, status, error) {
			        xhr.isHandled = true;
			        console.error("❌ ERROR HANDLER CALLED:", {
			            status: status,
			            error: error,
			            responseText: xhr.responseText,
			            timestamp: new Date().toISOString()
			        });
			        
			        $("#loader").removeClass('d-flex').addClass('d-none');
			        $("#statusBar").text('Error loading data.');
			        $("#startReceiving, #openSettings, #openPOD").html('Retry').prop('disabled', false).off('click').on('click', function () {
			            location.reload();
			        });
			    }
			});								

            $("#pending-receipts").click(function () {
                window.location.href = address + '/pendingReceipts';
            });

            $("#completed-today").click(function () {
                window.location.href = address + '/completedReceipts';
            });

            $('#startReceiving').click(function () {
                $("#loader").removeClass('d-none').addClass('d-flex').fadeIn();
                window.location.href = address + '/pods';
            });

            $('#openPOD').click(function () {
                $("#loader").removeClass('d-none').addClass('d-flex').fadeIn();
                window.location.href = address + '/newPod';
            });

            $('#openSettings').click(function () {
                $("#loader").removeClass('d-none').addClass('d-flex').fadeIn();
                window.location.href = address + '/users'; // Navigate to users.html for user management
            });

            // Initialize Bootstrap tooltips
            $('[data-bs-toggle="tooltip"]').tooltip();
        });
    </script>
